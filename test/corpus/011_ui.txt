var count = 10

fn inc_count() {
    count = count + 1
}

ui {
    text(width:5, value:"{count}")

    button(class:"primary", onclick:"inc_count") {
        text("+")
    }
}

style {
    ".primary": {
        color: "blue"
        "border-radius": "5px"
    }
}
